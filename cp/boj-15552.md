[15552 - 빠른 A+B](https://www.acmicpc.net/problem/15552)

```cpp
#include <bits/stdc++.h> 
using namespace std;
int main(void) {
  ios::sync_with_stdio(0);
  cin.tie(0);

  int t, a, b;
  scanf("%d", &t);

  for(int i=0; i<t; ++i) {
    scanf("%d %d", &a, &b);
    printf("%d\n", a+b);
  }

  return 0;
}
```
