#include <bits/stdc++.h>
using namespace std;

int main() {
  unsigned arr[41] = {0, 1, 1, 2, 3, 5};
  for(int i=6; i<41; ++i) 
    arr[i] = arr[i-2] + arr[i-1];

  int t;
  cin >> t;
  while(t--) {
    int a;
    cin >> a;
    if(a==0) cout << "1 0" << endl;
    else cout << arr[a-1] << ' ' << arr[a] << endl;
  }
  return 0;
}
