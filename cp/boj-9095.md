#include <bits/stdc++.h>
using namespace std;

int main(void) {
  int arr[11] = {1, 2, 4, 7};
  for(int i=4; i<11; ++i)
    arr[i] = arr[i-1] + arr[i-2] + arr[i-3];
  
  int t;
  cin >> t;
  while(t--) {
    int v; 
    cin >> v;
    cout << arr[v-1] << endl;
  }
  return 0;
}
